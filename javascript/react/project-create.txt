1. 项目创建， vite
    npm create vite@lastest

2. 环境设置
    npm i 
    npm i eslint vite-plugin-eslint eslint-config-react-app --save-dev

3. 文件设置
    创建eslintrc.json文件 (eslintrc.cjs已有，不需要)
        { 'extents': 'react-app' };
    修改vite.config.js文件
        import eslint from 'vite-plugin-eslint';
        export default defineConfig({
          plugins: [
            react(),
            eslint({
              failOnError: false, 
              failOnWarning: false,
            }),
          ],
        });
        
4. 删除文件
    App.css
    index.css
    assets文件夹
